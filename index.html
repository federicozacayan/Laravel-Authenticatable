Laravel Authenticatable

Install laravel/ui.

composer require laravel/ui --dev

Install javascript tools.

php artisan ui vue --auth

Run javascript tools.

npm install && npm run dev

Create a model

php artisan make:model Administrator --all

After configuring database run the migrations.

php artisan migrate

Copy and paste user model and change the class name and the file name by Administrator and app/Administrator.php.

class Administrator extends Authenticatable

Add a new guard in config/auth.php.

'guards' => [
    ...
    'admins' => [
        'driver' => 'session',
        'provider' => 'admins',
    ],
],

Add a new provider also in the same file.

'providers' => [
    ...
    'admins' => [
        'driver' => 'eloquent',
        'model' => App\Administrator::class,
    ],
],

Also the password section.

'passwords' => [
    ...
    'admins' => [
        'provider' => 'admins',
        'table' => 'password_resets',
        'expire' => 60,
        'throttle' => 60,
    ],
],

Add some routes in routes/web.php file.

Route::get('/admin/login', 'AdministratorController@showLoginForm');
Route::post('/admin/login', 'AdministratorController@login')->name('admin.login');
Route::get('/admin/area', 'AdministratorController@secret')->name('admin.area');

Route::get('/admin/password/reset','AdminForgotPasswordController@showLinkRequestForm')->name('admin.password.request');
Route::post('/admin/password/email','AdminForgotPasswordController@sendResetLinkEmail')->name('admin.password.email');

Route::get('/admin/password/reset/{token}','AdminResetPasswordController@showResetForm')->name('password.reset');
Route::post('/admin/password/reset','AdminResetPasswordController@reset')->name('password.update');

Route::get('/admin/password/confirm', 'AdministratorConfirmPasswordController@showConfirmForm')->name('admin.password.confirm');
Route::post('/admin/password/confirm', 'AdministratorConfirmPasswordController@confirm');

Create controller for reset password request.

php artisan make:controller AdminForgotPasswordController

app/Http/Controllers/AdminForgotPasswordController.php

<?php
namespace App\Http\Controllers\Auth;
use App\Http\Controllers\Controller;
use Illuminate\Foundation\Auth\SendsPasswordResetEmails;
class AdminForgotPasswordController extends Controller
{
    use SendsPasswordResetEmails;
    public function broker()
    {
        return Password::broker('admins');
    }
}

Create controller for reset password process.

php artisan make:controller AdministratorResetPasswordController

app/Http/Controllers/AdminResetPasswordController.php

<?php
namespace App\Http\Controllers\Auth;
use App\Http\Controllers\Controller;
use App\Providers\RouteServiceProvider;
use Illuminate\Foundation\Auth\ResetsPasswords;
class AdminResetPasswordController extends Controller
{
    use ResetsPasswords;
    protected $redirectTo = RouteServiceProvider::HOME;
    public function broker()
    {
        return Password::broker('admins');
    }
}

Create controller for confirm password reset process.

php artisan make:controller AdminConfirmPasswordController

Set up the controller in app/Http/Controllers/AdministratorController.php file.

<?php
namespace App\Http\Controllers\Auth;
use App\Http\Controllers\Controller;
use App\Providers\RouteServiceProvider;
use Illuminate\Foundation\Auth\ConfirmsPasswords;
class AdministratorConfirmPasswordController extends Controller
{
    use ConfirmsPasswords;
    protected $redirectTo = RouteServiceProvider::HOME;
    public function __construct()
    {
        $this->middleware('auth');
    }
    public function showConfirmForm()
    {
        return view('admin.confirm');
    }
}

Set up the controller in app/Http/Controllers/AdministratorController.php file.

<?php
namespace App\Http\Controllers;
use App\Administrator;
use Illuminate\Http\Request;
use Illuminate\Foundation\Auth\AuthenticatesUsers;
use Illuminate\Support\Facades\Auth;
class AdministratorController extends Controller
{
    public function __construct()
    {
        $this->middleware('auth:admins', ['only' => ['secret']]);
    }
    use AuthenticatesUsers;
    public function showLoginForm()
    {
        return view('admin.login');
    }
    private $redirectTo = 'admin/area';
    public function secret(){
        $user = auth('admins')->user();
        echo 'Admin id #'.$user->id;
    }
    protected function guard()
    {
        return Auth::guard('admins');
    }
}
